bug: 3d_main.html中抽屉打开，被点击节点应该处于除抽屉宽度后的区域中心，而现在它处在了绝对屏幕的中心。


正确处理：
- 使用简单的相机偏移算法

- 在 `handleNodeClick` 函数中：`{ x: node.x + (node.x * offsetRatio), y: node.y, z: node.z }`

- 在 `onNodeClick` 回调中：`{ x: node.x + (Math.abs(node.x) + 200) * offsetRatio, y: node.y, z: node.z }`

- 偏移系数：`offsetRatio = (DRAWER_WIDTH / screenWidth) * 0.7`

- 没有使用 `setCameraViewOffset` 函数

bug: 3d_main.html中抽屉打开，被点击节点应该处于除抽屉宽度后的区域中心，而现在它漂移的，好像不固定。

这个问题出现的核心原因在于：当抽屉打开时，3D 场景的渲染画布（Canvas）依然占据全屏，但你希望视觉中心偏移到“全屏宽度 - 抽屉宽度”的中心。

在 3D 空间中，简单的坐标平移（x + offset）会随着相机距离（Zoom）的变化而导致“漂移”。要实现精确的“区域居中”，必须根据相机的 **FOV（视场角）**和 距离 动态计算偏移量。

以下是一些修复后的关键要点，供参考
```javascript
/**
 * 元龙检索 - 因果星空 (Causal Starry Sky) 全功能集成驱动 v2.3
 * 职责：3D 可视化引擎、精确中心偏移校准、因果链溯源、节点管理
 */

// --- [1. 全局状态管理] ---
let Graph = null;
const highlightNodes = new Set();
const highlightLinks = new Set();
let hoverNode = null;
let currentSelectedNodeId = null; 
let is_change = true; // true: 编辑模式, false: 链入父ID模式
let nodeCache = {}; 

// 巡航与控制状态
let isPhysicsEnabled = true;
let isDragonCruising = false;
let selectedNodeObj = null; 

// 常量定义
const DRAWER_WIDTH = 384; // 抽屉宽度 (w-96 = 24rem = 384px)
const FOCUS_DIST = 200;   // 聚焦距离

// --- [2. 核心辅助工具] ---

function getThreeInstance() {
    return window.THREE || (Graph && Graph.scene() ? Graph.scene().__threeObj : null);
}

/** 视觉反馈：在界面显示简短提示 */
function showSelectionHint(msg) {
    const hint = document.getElementById('selection-hint');
    if (!hint) return;
    hint.textContent = msg;
    hint.classList.remove('opacity-0');
    setTimeout(() => hint.classList.add('opacity-0'), 2000);

// --- [3. 关键修复：精确坐标偏移计算] ---
/**
 * 计算带偏移的相机位置和观察点
 * 目的是让节点显示在 (屏幕宽度 - 抽屉宽度) 的中心
 */
function calculateOffsetView(node, distance = FOCUS_DIST) {
    const camPos = Graph.cameraPosition();
    const { x, y, z } = node;

    // 1. 计算从原点到节点的单位向量（方向）
    const dist = Math.hypot(x, y, z) || 1;
    const dir = { x: x / dist, y: y / dist, z: z / dist };

    // 2. 目标相机位置（在节点方向上向外延伸固定距离）
    const newCamPos = {
        x: x + dir.x * distance,
        y: y + dir.y * distance,
        z: z + dir.z * distance
    };

    // 3. 计算水平偏移量
    // 原理：在 3D 空间中，我们需要将 lookAt 点向右偏移，使节点在屏幕上向左移动
    const screenWidth = window.innerWidth;
    const drawerVisible = !document.getElementById('drawer').classList.contains('drawer-hidden');
    
    let lookAtPos = { x, y, z };

    if (drawerVisible && screenWidth > DRAWER_WIDTH) {
        // 计算抽屉占屏幕的比例
        const offsetRatio = DRAWER_WIDTH / screenWidth;
        
        // 基于相机距离和FOV计算世界坐标系下的偏移宽度
        // 这里的 0.65 是一个经验补偿系数，确保节点位于左侧区域的几何中心
        const fovRad = (Graph.camera().fov * Math.PI) / 180;
        const visibleWidthAtDist = 2 * Math.tan(fovRad / 2) * distance;
        const worldOffset = visibleWidthAtDist * offsetRatio * 0.5;

        // 获取相机右向量（Right Vector），使偏移垂直于视线
        const camera = Graph.camera();
        const rightVec = new THREE.Vector3(1, 0, 0).applyQuaternion(camera.quaternion);

        lookAtPos = {
            x: x + rightVec.x * worldOffset,
            y: y + rightVec.y * worldOffset,
            z: z + rightVec.z * worldOffset
        };
    }

    return { camPos: newCamPos, lookAt: lookAtPos };
}

// --- [4. 引擎初始化] ---

function initGraph() {
    const elem = document.getElementById('3d-graph');
    Graph = ForceGraph3D()(elem)
        .backgroundColor('rgba(0,0,0,0)')
        .nodeLabel(node => {
            // 遵循术语规范：职责与业务
            return `
                <div class="force-graph-tooltip">
                    <div class="tooltip-title">${node.node_id}</div>
                    <div class="tooltip-row">
                        <span class="tooltip-label">职责:</span>
                        <span class="tooltip-value">${node.duty || '未定义'}</span>
                    </div>
                    <div class="tooltip-row">
                        <span class="tooltip-label">权重:</span>
                        <span class="tooltip-value weight-value">${(node.survival_weight || 0).toFixed(4)}</span>
                    </div>
                </div>
            `;
        })
        .nodeColor(node => {
            if (highlightNodes.has(node.node_id)) return node.node_id === currentSelectedNodeId ? '#f59e0b' : '#60a5fa';
            return node.color || '#1d4ed8';
        })
        .nodeRelSize(6)
        .nodeVal(node => Math.max(0.1, node.survival_weight || 0.1))
        .linkWidth(link => highlightLinks.has(link) ? 4 : 1)
        .linkColor(link => highlightLinks.has(link) ? '#60a5fa' : '#334155')
        .linkDirectionalArrows(2)
        .linkDirectionalArrowLength(4)
        .onNodeClick(node => {
            if (is_change) {
                focusOnNode(node);
                openDrawer(node.node_id);
            } else {
                // 职责链链入模式
                linkToParent(node.node_id);
            }
        })
        .onNodeHover(node => {
            if (node === hoverNode) return;
            hoverNode = node;
            highlightNodes.clear();
            highlightLinks.clear();
            if (node) {
                highlightNodes.add(node.node_id);
                // 可以在这里增加邻居节点高亮逻辑
            }
            updateStyles();
        });

    // 初始物理仿真参数
    Graph.d3Force('charge').strength(-120);
    Graph.d3Force('link').distance(150);
}

function updateStyles() {
    Graph.nodeColor(Graph.nodeColor())
         .linkWidth(Graph.linkWidth())
         .linkColor(Graph.linkColor());
}

// --- [5. 交互业务逻辑] ---

function focusOnNode(node) {
    selectedNodeObj = node;
    currentSelectedNodeId = node.node_id;
    
    const { camPos, lookAt } = calculateOffsetView(node);
    
    Graph.cameraPosition(
        camPos, 
        lookAt,
        1000 // 1秒平滑过渡
    );
}

function openDrawer(nodeId) {
    const drawer = document.getElementById('drawer');
    const data = nodeCache[nodeId];
    if (!data) return;

    // 填充内容
    document.getElementById('d-node-id').value = nodeId;
    document.getElementById('d-event-tuple').value = data.event_tuple || '';
    
    // 更新职责状态条
    const weight = data.survival_weight || 0;
    document.getElementById('d-weight-bar').style.width = `${weight * 100}%`;
    document.getElementById('d-weight-value').textContent = parseFloat(weight).toFixed(8);

    drawer.classList.remove('drawer-hidden');
    
    // 抽屉打开后，延时重新触发一次对齐校准
    setTimeout(() => {
        if (selectedNodeObj) focusOnNode(selectedNodeObj);
    }, 100);
}

function closeDrawer() {
    document.getElementById('drawer').classList.add('drawer-hidden');
    if (selectedNodeObj) {
        // 抽屉关闭，回到屏幕正中心
        const { camPos, lookAt } = calculateOffsetView(selectedNodeObj);
        Graph.cameraPosition(camPos, selectedNodeObj, 800);
    }
}

// --- [6. 响应式与自适应] ---

window.addEventListener('resize', () => {
    if (!Graph) return;
    Graph.width(window.innerWidth).height(window.innerHeight);
    
    // 如果抽屉开着，调整相机以保持节点在左侧中心
    if (selectedNodeObj && !document.getElementById('drawer').classList.contains('drawer-hidden')) {
        const { camPos, lookAt } = calculateOffsetView(selectedNodeObj);
        Graph.cameraPosition(camPos, lookAt, 0); // 瞬间调整不使用动画
    }
});

// --- [7. 数据驱动业务] ---

async function loadHistory() {
    try {
        const response = await fetch('/api/v1/causal/history');
        const res = await response.json();
        if (res.status === 'success') {
            const gData = {
                nodes: res.data.map(d => ({ ...d, id: d.node_id })),
                links: []
            };
            
            // 构建因果链
            res.data.forEach(n => {
                nodeCache[n.node_id] = n;
                const parents = n.parent_ids || (n.parent_id ? [n.parent_id] : []);
                parents.forEach(p => {
                    gData.links.push({ source: p, target: n.node_id });
                });
            });
            
            Graph.graphData(gData);
        }
    } catch (e) {
        console.error("加载历史数据失败:", e);
    }
}
```
